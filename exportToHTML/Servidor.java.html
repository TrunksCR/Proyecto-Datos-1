121      public LinkedList(){
122          this.head=null;
123          this.size=0;
124          this.tail=null;
125      }
126      public int size() {
127          return this.size;
128      }
129      public void insert(Object data){
130          if(head==null){
131              head=tail= new Node(data);
132          }else{
133              tail=tail.next = new Node(data);
134          }
135      }
